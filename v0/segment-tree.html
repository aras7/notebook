typedef int NUM;
class SegmentTree
{
     NUM *T;
     
     public:
     
     SegmentTree(){}
     SegmentTree(int n)
     {
         int size = 2 * potencia(2 , (static_cast<int>(log2(n)) + 1)) + 7;

         T = new int[size];

         for(int i = 0;i < size;i++)
             T[i] = -1;
     }
     
     void Initialize(int node, int b,int e, NUM *array)// tree.Initialize( 1,0,N-1,arr);
     {
         if(b == e)
             T[node] = b;
         else
         {
              int x = (b + e) / 2;
              Initialize(2 * node, b , x , array);
              Initialize(2 * node + 1, x+1 , e , array);
              
              if(array[ T[2 * node] ] >= array[ T[2 * node + 1] ] )
                  T[node] = T[2 * node];
              else
                  T[node] = T[2 * node + 1]; 
          }
     }
     
     // it returns the position
     NUM Query(int node,int b,int e,int i,int j, NUM *array)// tree.Query(1,0,N-1,i,j,arr)
     {
          if(i > e || j < b)
              return -1;

          if(b >= i && e <= j)
              return T[node];

          int r[2],x = (b + e) / 2;
          
          r[0] = Query(2 * node, b, x, i , j , array) ,
          r[1] = Query(2 * node + 1, x+1, e , i , j , array); 
       
          if(r[0] == -1)
              return r[1];
          if(r[1] == -1)
              return r[0];
              
          return array[r[0]] >= array[r[1]] ? r[0] : r[1];
     }
};
