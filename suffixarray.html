#define TAM (2000)
     
char S[TAM];
int length,sum_lcp,*suffix_index,*lcp;

bool CMP(int a,int b)
{
     if(a == b)return false;
     
     for(int i = a,j = b;;i++,j++)
         if(S[i] != S[j])
            return S[i] < S[j];
            
     return false;
}

class SuffixArray
{
     public:
     
     SuffixArray(){}
     
     void Create()
     {
          length = strlen(S);
          S[length++] = '$';
          
          suffix_index = new int[length];
          lcp = new int[length];
          
          for(int i = 0;i < length;i++)
             lcp[i] = 0, suffix_index[i] = i;
             
          sum_lcp = 0;
          FillSuffix();
          FillL_LCP();
     }
     
     void FillSuffix()
     {
          sort(suffix_index,suffix_index + length,CMP);
     }
     
     void FillL_LCP()
     {
          sum_lcp = 0;
          for(int c,j,k,i = 0,__length = length - 1;i < __length;i++)
          {
              for(c = 0,j = suffix_index[i],k = suffix_index[i + 1]; ;j++,k++,c++)
                  if(S[j] != S[k])
                      break;
              
              lcp[i] = c;
              sum_lcp += c;
          }
     }
     
};
              
SuffixArray suffix_array;
scanf("%s",S);
suffix_array.Create();
