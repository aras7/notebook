long long Comb(int a,int b)
{
     long long total = 1;
     int i = a,j = 2,limit = a-b;
     if(b < limit)b = limit,limit = a-b;
     while(i > b)
     {
         total *= i--;
         while(!(total % j) && j <= limit)total /= j++;
     }
     return total;
}
