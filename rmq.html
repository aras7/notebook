typedef int NUM;
vector > RMQ(vector X)
{
     int N = X.size();
     
     vector >M ( N, vector (static_cast(log2(N))+1,0 ));
     
     for(int i = 0;i < N;i++)
        M[i][0] = i;
        
     for(int h,i = 1; (1 << i) <= N; i++)
         for(h = 0;h + (1 << i) - 1 < N;h++)
             if( X[ M[h][i-1] ] >= X[ M[h + (1 << (i-1))][i-1] ])
                 M[h][i] = M[h][i-1];
             else
                 M[h][i] = M[h + (1 << (i-1))][i-1];
     
     return M;
}
